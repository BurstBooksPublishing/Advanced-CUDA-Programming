__global__ void highArithmeticIntensityKernel(float* A, float* B, float* C, int N) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < N) {
        float sum = 0.0f;
        // Perform multiple operations to increase arithmetic intensity
        for (int i = 0; i < 100; ++i) {
            sum += A[idx] * B[idx] + A[idx] * C[idx] + B[idx] * C[idx];
        }
        C[idx] = sum;
    }
}
