#include 
#include 

__global__ void kernel(int *d_data, int N) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < N) {
        d_data[idx] = idx * 2; // Simple computation
    }
}

int main() {
    const int N = 1024;
    int *h_data = new int[N];
    int *d_data;
    cudaMalloc(&d_data, N * sizeof(int));

    // Launch kernel with default stream
    kernel<<<(N + 255) / 256, 256>>>(d_data, N);

    // Synchronize CPU and GPU using cudaDeviceSynchronize
    cudaDeviceSynchronize(); // Ensures kernel execution is complete

    // Copy data back to host
    cudaMemcpy(h_data, d_data, N * sizeof(int), cudaMemcpyDeviceToHost);

    // Verify the result
    for (int i = 0; i < N; ++i) {
        if (h_data[i] != i * 2) {
            std::cerr << "Error at index " << i << std::endl;
            break;
        }
    }

    // Cleanup
    delete[] h_data;
    cudaFree(d_data);

    return 0;
}
