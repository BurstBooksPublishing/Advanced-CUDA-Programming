#define N 1024  // Size of the array

__global__ void vectorAdd(int *A, int *B, int *C, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        C[idx] = A[idx] + B[idx];
    }
}

int main(int argc, char **argv) {
    int rank, size;
    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    int *h_A, *h_B, *h_C;
    int *d_A, *d_B, *d_C;
    int chunk_size = N / size;

    // Allocate host memory
    h_A = (int *)malloc(chunk_size * sizeof(int));
    h_B = (int *)malloc(chunk_size * sizeof(int));
    h_C = (int *)malloc(chunk_size * sizeof(int));

    // Initialize host arrays
    for (int i = 0; i < chunk_size; i++) {
        h_A[i] = i + rank * chunk_size;
        h_B[i] = i + rank * chunk_size;
    }

    // Allocate device memory
    cudaMalloc((void **)&d_A, chunk_size * sizeof(int));
    cudaMalloc((void **)&d_B, chunk_size * sizeof(int));
    cudaMalloc((void **)&d_C, chunk_size * sizeof(int));

    // Copy data to device
    cudaMemcpy(d_A, h_A, chunk_size * sizeof(int), cudaMemcpyHostToDevice);
    cudaMemcpy(d_B, h_B, chunk_size * sizeof(int), cudaMemcpyHostToDevice);

    // Launch kernel
    int threadsPerBlock = 256;
    int blocksPerGrid = (chunk_size + threadsPerBlock - 1) / threadsPerBlock;
    vectorAdd<<>>(d_A, d_B, d_C, chunk_size);

    // Copy result back to host
    cudaMemcpy(h_C, d_C, chunk_size * sizeof(int), cudaMemcpyDeviceToHost);

    // Gather results from all processes
    int *global_C = NULL;
    if (rank == 0) {
        global_C = (int *)malloc(N * sizeof(int));
    }
    MPI_Gather(h_C, chunk_size, MPI_INT, global_C, chunk_size, MPI_INT, 0, MPI_COMM_WORLD);

    // Print result from root process
    if (rank == 0) {
        for (int i = 0; i < N; i++) {
            std::cout << global_C[i] << " ";
        }
        std::cout << std::endl;
        free(global_C);
    }

    // Free memory
    free(h_A);
    free(h_B);
    free(h_C);
    cudaFree(d_A);
    cudaFree(d_B);
    cudaFree(d_C);

    MPI_Finalize();
    return 0;
}
